On both machines:
=================
# Installing docker 
sudo apt-get update
sudo apt-get upgrade -y
curl -sSL https://get.docker.com/ | sh

# Add the user that will carry out the tests to the docker group
usermod -aG docker (username)

# Adding both machines to the hosts file so that they can be found by name 
sudo echo (ip of the driver) driver >> /etc/hosts
sudo echo (ip of the testbed) testbed >> /etc/hosts

Only driver:
============
# Python installation
apt install -y python3.8 python3.8-dev
sudo update-alternatives --install /usr/bin/python python /usr/bin/python3.8 1
sudo update-alternatives --set python /usr/bin/python3.8

# Installion of Jupyter Notebook
cd ~
wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh -nv -O /tmp/miniconda.sh
bash /tmp/miniconda.sh -b -p ~/miniconda

echo export PATH=~/miniconda/bin:$PATH >> ~/.bashrc
source ~/.bashrc

eval "$(~/miniconda/bin/conda shell.bash hook)"
conda init
conda install -c r r-essentials -y
conda install -c conda-forge notebook -y
conda install -c anaconda python=3.8 pip -y
pip install requests locust psutil docker

#cp -r /vagrant/configuration/jupyter ~/.jupyter

# Update to the last version of Git and configure it
sudo add-apt-repository ppa:git-core/ppa -y
sudo apt-get update
sudo apt-get install git -y

# Docker swarm installation and writing join token into file
docker swarm init --advertise-addr (ip of the driver) --listen-addr (ip of the driver)
